module LFSR(
input wire clock,
input wire reset
output reg [7:0] lfsr
);

localparam TAPS = 8'b10101010;

wire x, feedback;

integer i1, i2;

assign feedback = ~| lfsr[6:0] ^ lfsr[7];

always @(posedge clock or negedge reset)
begin
	if(!reset)
	begin
		lfsr <= 8'b0;
	end
	else
	begin
		lfsr[0] <= feedback;
		
		for (i = 1; i < 8; i = i + 1)
		begin
			if(TAPS[i])
				lfsr[i] <= lfsr[i - 1] ^ feedback;
			else
				lfsr[i] <= lfsr[i - 1];
		end
	end
end

endmodule
